# Atividade-De-Classe-Python3
class PACE :

    def __init__(self, tempo, distancia):

        self.tempo = tempo

        self.distancia = distancia

    def setTempo(self, tempo):

        self.tempo = tempo

    def setDistancia(self, distancia):

        self.distancia = distancia

    def getTempo(self):

        return self.tempo
    def getDistancia(self):

        return self.distancia
